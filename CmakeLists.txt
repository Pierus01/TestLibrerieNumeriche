cmake_minimum_required(VERSION 3.18)
project(TestEfficienza VERSION 1.0)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

configure_file(TestEfficienzaConfig.h.in TestEfficienzaConfig.h)

add_subdirectory(librerie)

list(APPEND EXTRA_LIBS parabola)
list(APPEND EXTRA_LIBS integrale)

add_executable(t_efficienza test_efficienza.cxx)

target_link_libraries(t_efficienza PUBLIC ${EXTRA_LIBS})
target_include_directories(t_efficienza PUBLIC "${PROJECT_BINARY_DIR}")
